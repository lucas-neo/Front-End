
# Exibição de dados

# 3 formas de declarar variaval 

    -> Var
    -> Let
    -> Const

    let nome = valor;

    const nome = valor;
    - Sempre use const se o valor não precisar ser alterado
    - Sempre use const se o tipo não de deve ser alterado (arrays e  objeto)
